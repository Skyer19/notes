
python怎样判断一个目录中的文件有哪些
<pre>
    directory = model_dir
    entries = os.listdir(directory)
    # 筛选出所有文件，忽略文件夹
    files = [entry for entry in entries if os.path.isfile(os.path.join(directory, entry))]
    
    # 打印文件列表
    print(files)
